using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Xml.Serialization;

namespace LibrarySystem
{
    internal class librarian : Member
    {
        public void AddBook(Book b, Library lib1) 
        {
            lib1.Books.Add(b);
            Console.WriteLine($"{b.Title} has been added to the library");
        }
        public void RemoveBook(Book b, Library lib)
        {
            lib.Books.Remove(b);
            Console.WriteLine($"{b.Title} has been removed from the library.");
        }
    }
    }
